```ts
// Lint runs automated design quality and accessibility checks on a node tree.
// Rules: "all" (default), "wcag" (accessibility only), or specific rule names. Use help(topic:"check") for full rule list.
// ---
// Rules:
//   "all" (default) — run every rule. "wcag" — all accessibility rules only.
//   Design rules:
//     "no-autolayout" — frames with manually positioned children (should use auto-layout)
//     "shape-instead-of-frame" — shapes used as containers (should be frames)
//     "hardcoded-color" — colors not using styles or variables (should use design tokens)
//     "no-text-style" — text without a text style applied
//     "fixed-in-autolayout" — FIXED-size children in auto-layout parents (should be HUG or FILL)
//     "default-name" — nodes with default names like "Frame 1" (should be renamed)
//     "empty-container" — empty frames/groups (likely leftover, should be deleted)
//     "stale-text-name" — text node name doesn't match its content
//     "no-text-property" — component text not exposed as editable property
//   Accessibility (WCAG) rules:
//     "wcag-contrast" — text contrast ratio below 4.5:1 (WCAG AA)
//     "wcag-contrast-enhanced" — text contrast ratio below 7:1 (WCAG AAA)
//     "wcag-non-text-contrast" — non-text element contrast below 3:1
//     "wcag-target-size" — interactive targets smaller than 24x24px (WCAG 2.5.8)
//     "wcag-text-size" — text below 12px
//     "wcag-line-height" — line height below 1.5x font size
// maxDepth limits how deep the tree traversal goes (default: 10). maxFindings caps total findings (default: 50).
// fix: auto-converts frames to auto-layout. Use after lint.check identifies "no-autolayout" issues.
interface LintResult {
  nodeId: string; nodeName: string;
  categories: { rule: string; count: number; fix: string; nodes: { id: string; name: string; detail?: string }[] }[];
  warning?: string;
}
```
